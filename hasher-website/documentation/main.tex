\section{Założenia projektowe}
\subsection{Założenia techniczne i nietechniczne}
	\begin{itemize}
	    \item \textbf{Założenia nietechniczne}
            \begin{itemize}
                \item Możliwość wyboru algorytmu kodowania
                \item Możliwość szyfrowania/deszyfrowania danych
                \item Możliwość wyboru algorytmu haszowania
                \item Możliwość haszowania danych
            \end{itemize}
        \item \textbf{Założenia techniczne}
            \begin{itemize}
                \item Aplikacja składa się z strony internetowej (frontendu) i serwera, który ja obsługuje
                \item Nie ma potrzeby tworzenia backendu i API
            \end{itemize}
	\end{itemize}

\subsection{Stos technologiczny}
	\begin{itemize}
	    \item \textbf{Narzędzia związane z zarządzaniem projektem}\\
            \textbf{Linear} - narzędzie do rozdzielania zadań i nadzorowania ich realizacji
        \item \textbf{Narzędzia wykonujące części składowe projektu}\\
            \textbf{Vite} - TypeScript połączony z React, służy do pisania stron internetowych\\
            \textbf{Serwer} - obsługuje stronę
	\end{itemize}

\subsection{Oczekiwane rezultaty projektu}
    \begin{itemize}
        \item Utworzenie ładnego interjfejsu przyjaznego użytkownikowi
        \item Stworzenie serwera obsługującego całą logikę
        \item Strona ma służyć do szyfrowania i haszowania używając różnych algorytmów.
    \end{itemize}
